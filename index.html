Information

Package sprock
Description Proxies for v8
Node Version >= 0.6.14

sprock is a native implementation of proxies

Usage

Sprock = require 'sprock'

obj =
  hello: 'test'
  other: 'tazt'

trap = (name) ->
  if name is 'special'
    return 'wot'
  else
    return @[name]

prox = new Sprock obj, trap

console.log prox.hello
console.log prox.special
console.log prox.waaaa

###
Output:
"test"
"wot"
undefined

###
